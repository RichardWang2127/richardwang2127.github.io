📘 第三课时：选择结构

ifswitch

🔹 if 结构 (与C++类似)

// 单分支
int age = 20;
if (age >= 18) {
    System.out.println("成年人");
}

// 双分支
int score = 85;
if (score >= 60) {
    System.out.println("及格");
} else {
    System.out.println("不及格");
}

// 多分支
int grade = 85;
if (grade >= 90) {
    System.out.println("优秀");
} else if (grade >= 80) {
    System.out.println("良好");
} else if (grade >= 60) {
    System.out.println("及格");
} else {
    System.out.println("不及格");
}
⚠️ Java中条件表达式必须为 boolean，if(1) 编译错误！

🔸 switch 传统写法

int week = 3;
switch (week) {
    case 1: System.out.println("星期一"); break;
    case 2: System.out.println("星期二"); break;
    case 3: System.out.println("星期三"); break;
    default: System.out.println("无效数字");
}

// 穿透特性 (无break)
char season = '夏';
switch (season) {
    case '春':
    case '夏':
    case '秋': System.out.println("适合旅游"); break;
    case '冬': System.out.println("适合宅家");
}

✨ Java 12+ 箭头语法

int week = 3;
String day = switch (week) {
    case 1 -> "星期一";
    case 2 -> "星期二";
    case 3 -> "星期三";
    default -> "无效";
};
System.out.println(day); // 星期三

📘 第四课时：循环结构

while / do-while / for

// while 打印1~5
int i = 1;
while (i <= 5) {
    System.out.println(i);
    i++;
}

// do-while 至少执行一次
int j = 1;
do {
    System.out.println(j);
    j++;
} while (j <= 5);   // ⚠️ 分号

// for 1~100求和
int sum = 0;
for (int k = 1; k <= 100; k++) {
    sum += k;
}
System.out.println("1+...+100 = " + sum); // 5050

break / continue

// 跳过3的倍数，到8停止
for (int i = 1; i <= 10; i++) {
    if (i % 3 == 0) continue;
    if (i == 8) break;
    System.out.println(i); // 1,2,4,5,7
}

📘 第五课时：二重循环

九九乘法表 & 直角三角形

// 乘法表
for (int i = 1; i <= 9; i++) {
    for (int j = 1; j <= i; j++) {
        System.out.print(j + "×" + i + "=" + (i*j) + "\t");
    }
    System.out.println();
}

// 打印直角三角形
int rows = 4;
for (int i = 1; i <= rows; i++) {
    for (int j = 1; j <= i; j++) {
        System.out.print("*");
    }
    System.out.println();
}

带标签的 break

outer:
for (int i = 0; i < 5; i++) {
    for (int j = 0; j < 5; j++) {
        if (j == 2) break outer; // 跳出外层
        System.out.println("i=" + i + ", j=" + j);
    }
}

📘 第六课时：一维数组

// 声明与初始化
int[] arr1 = new int[5];          // 默认0
int[] arr2 = {1,2,3,4,5};         // 静态初始化

// 遍历
int[] scores = {85,92,78,90,88};
for (int i = 0; i < scores.length; i++) {
    System.out.println(scores[i]);
}
for (int score : scores) {        // 增强for
    System.out.println(score);
}

// 求最大值
int[] numbers = {23,45,12,67,34,89};
int max = numbers[0];
for (int i = 1; i < numbers.length; i++) {
    if (numbers[i] > max) max = numbers[i];
}
System.out.println("最大值：" + max);
📌 数组默认值：整数0，浮点0.0，布尔false，引用null。

📘 第七课时：二维数组

// 定义与初始化
int[][] matrix = {
    {1,2,3},
    {4,5,6},
    {7,8,9}
};

// 遍历
for (int i = 0; i < matrix.length; i++) {
    for (int j = 0; j < matrix[i].length; j++) {
        System.out.print(matrix[i][j] + " ");
    }
    System.out.println();
}

// 不规则数组（杨辉三角）
int rows = 5;
int[][] yangHui = new int[rows][];
for (int i = 0; i < rows; i++) {
    yangHui[i] = new int[i+1];
    yangHui[i][0] = yangHui[i][i] = 1;
    for (int j = 1; j < i; j++) {
        yangHui[i][j] = yangHui[i-1][j-1] + yangHui[i-1][j];
    }
}

📘 第九课时：类的方法

方法定义/重载/构造/this

// 方法重载
public class Printer {
    public void print(int num) { }
    public void print(String str) { }
    public void print(String str, int times) { }
}

// 构造方法 & this
public class Person {
    String name;
    int age;
    public Person() {
        this("匿名", 18);  // 调用另一个构造器
    }
    public Person(String name, int age) {
        this.name = name;
        this.age = age;
    }
}

// 值传递 (引用类型修改内容)
public void changeArray(int[] arr) {
    if (arr != null) arr[0] = 100;  // 原数组改变
}

📘 第十课时：抽象和封装

访问修饰符 / getter/setter

public class Student {
    private String name;
    private int age;

    public String getName() { return name; }
    public void setName(String name) {
        if (name != null && !name.trim().isEmpty())
            this.name = name;
    }

    public int getAge() { return age; }
    public void setAge(int age) {
        if (age >= 0 && age <= 150) this.age = age;
    }
}

📘 第十一课时：类的继承

// 父类
class Animal {
    protected String name;
    public void eat() { System.out.println(name + "吃东西"); }
}

// 子类
class Dog extends Animal {
    public void bark() { System.out.println(name + "汪汪叫"); }

    @Override
    public void eat() { System.out.println(name + "啃骨头"); }
}

// 构造器链
public Dog(String name) {
    super(name);   // 必须第一行
}

// super 访问父类成员
super.eat();

📘 第十二课时：多态

// 父类引用指向子类对象
Animal a = new Dog("旺财");
a.eat();               // 调用Dog的eat

// 向下转型
if (a instanceof Dog) {
    Dog d = (Dog) a;
    d.bark();
}

// 多态应用
public void feed(Animal animal) {
    animal.eat();      // 动态绑定
}

📘 第十三课时：接口

interface Flyable {
    int MAX_HEIGHT = 10000;   // public static final
    void fly();               // public abstract
    default void show() { System.out.println("飞行器"); }
    static void info() { System.out.println("接口静态方法"); }
}

interface Eatable { void eat(); }

class Bird implements Flyable, Eatable {
    public void fly() { System.out.println("鸟儿飞"); }
    public void eat() { System.out.println("鸟儿吃"); }
}

📘 第十四课时：异常

try {
    int[] arr = new int[5];
    arr[10] = 100;
} catch (ArrayIndexOutOfBoundsException e) {
    System.out.println("索引越界:" + e);
} catch (Exception e) {   // 父类放最后
    e.printStackTrace();
} finally {
    System.out.println("总会执行");
}

// throws 声明异常
void readFile() throws IOException { ... }

// throw 手动抛出
throw new IllegalArgumentException("参数错误");

// 自定义异常
class MyException extends Exception {
    public MyException(String msg) { super(msg); }
}

📘 第十五课时：集合框架

ArrayList / HashSet / HashMap

// List
List<String> list = new ArrayList<>();
list.add("Java"); list.add("Python");
for (String s : list) System.out.println(s);

// Set  (去重)
Set<Integer> set = new HashSet<>();
set.add(10); set.add(10);  // 只有一个10

// Map
Map<String, Integer> map = new HashMap<>();
map.put("张三", 85);
System.out.println(map.get("张三"));  // 85

// 遍历
for (Map.Entry<String,Integer> e : map.entrySet()) {
    System.out.println(e.getKey() + ":" + e.getValue());
}

📘 第十六课时：实用类 (String, StringBuilder...)

// String 不可变
String s = "  Hello  ";
System.out.println(s.trim());         // "Hello"
System.out.println(s.substring(2,5)); // "ell"

// StringBuilder 可变, 单线程高效
StringBuilder sb = new StringBuilder("Hello");
sb.append(" World").insert(5, " Java");
System.out.println(sb);  // Hello Java World

// Math
Math.max(10,20); Math.min(5,8); Math.random();

// Arrays
int[] arr = {3,1,4};
Arrays.sort(arr);
System.out.println(Arrays.toString(arr)); // [1,3,4]
